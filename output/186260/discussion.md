## Body

small.c

```c
static int h = 1;
int *c[][0][0];
static int *b();
static unsigned int g() {
  int *c = &h;
  b(*c);
}
int *b(unsigned int a) {
  int f = 1342177280;
  int **b = &c[0][0][0];
  int d, e;
  int *g = &e;
  for (;;) {
    if (e)
      c[a][a][a] = 0;
    else
      *g = a;
    if (a) {
      int c = 10, h = e;
      for (; h < c; h++) {
        *b = g;
        for (d = 0; d != -10; --d) {
          a = f + 1000000000;
          for (int h = a; h < e; h++)
            *b = &f;
        }
      }
    } else
      continue;
    int *d = &h;
    return d;
  }
}
int main() { g(); }
```

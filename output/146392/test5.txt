static inline __attribute__((__gnu_inline__))
__attribute__((no_instrument_function)) bool
folio_has_pincount(const struct folio *folio) {
  if (1)
    return folio_test_large(folio);
  return folio_order(folio) > 1;
}

static inline __attribute__((__gnu_inline__))
__attribute__((no_instrument_function)) bool
folio_maybe_dma_pinned(struct folio *folio) {
  if (folio_has_pincount(folio))
    return atomic_read(&folio->_pincount) > 0;
  return ((unsigned int)folio_ref_count(folio)) >= (1U << 10);
}
